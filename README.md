# epostit

# 1. nojs & npm 설치
node js 설치 필요
node, npm 버전확인

# 2. vue 설치
$ npm install vue

# 3. vue-cli 설치
$ npm install --global vue-cli

# 4. ePostIT\src\front node 모듈 설치
$ npm install

# ePostIT\src\front 폴더에서 vue 프로젝트 빌드
$ npm run build
